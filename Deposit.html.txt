  <script>
    function copyText(id) {
      const text = document.getElementById(id).textContent;
      navigator.clipboard.writeText(text).then(() => {
        showToast("Copied: " + text);
      });
    }

    function showToast(message) {
      const toast = document.getElementById("toast");
      toast.textContent = message;
      toast.className = "toast show";
      setTimeout(() => {
        toast.className = toast.className.replace("show", "");
      }, 2500);
    }

    function updateDeposit() {
      const amount = document.getElementById("amountInput").value || "1171";
      const blockchain = document.getElementById("blockchainInput").value || "TRC20";
      const type = document.getElementById("typeInput").value || "Deposit";
      const status = document.getElementById("statusInput").value || "Received";
      const address = document.getElementById("addressInput").value || "TNxxxxxxx";
      const txid = document.getElementById("txidInput").value || "s1tz2q7...ir8r";
      const time = document.getElementById("timeInput").value || new Date().toLocaleString();

      document.getElementById("amount").innerHTML = `+${amount} USDT <span class="received">? ${status}</span>`;
      document.getElementById("blockchain").textContent = blockchain;
      document.getElementById("type").textContent = type;
      document.getElementById("status").textContent = status;
      document.getElementById("depositAddress").textContent = address;
      document.getElementById("txid").textContent = txid;
      document.getElementById("time").textContent = time;

      // Update explorer link
      document.getElementById("explorerLink").href = `https://tronscan.org/#/transaction/${txid}`;
    }
  </script>
